using MySql.Data.MySqlClient;
using System;

namespace MySQLConnectionExample
{
    class Program
    {
        static void Main(string[] args)
        {
            // Define the connection string
            string connectionString = "Server=localhost;Port=3306;Database=TestDb;User=root;Password=yourpassword;";

            // Create a MySQL connection object
            using (var connection = new MySqlConnection(connectionString))
            {
                try
                {
                    // Open the connection to the MySQL server
                    connection.Open();
                    Console.WriteLine("Connection successful!");

                    // Run a simple query (e.g., SELECT)
                    string query = "SELECT Id, Name, Age FROM Users;";
                    using (var cmd = new MySqlCommand(query, connection))
                    using (var reader = cmd.ExecuteReader())
                    {
                        // Loop through the results and print them
                        while (reader.Read())
                        {
                            int id = reader.GetInt32("Id");
                            string name = reader.GetString("Name");
                            int age = reader.GetInt32("Age");
                            Console.WriteLine($"ID: {id}, Name: {name}, Age: {age}");
                        }
                    }
                }
                catch (Exception ex)
                {
                    Console.WriteLine($"Error: {ex.Message}");
                }
            }
        }
    }
}
